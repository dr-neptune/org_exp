 #+BEGIN_SRC R :results value :colnames yes :hline yes
 library(tidyverse)
 library(magrittr)
 set.seed(8889)
 
 x <- 1:10
 x <- x * x * x
 y <- runif(n = 10, max = 0.5)
 z <- rcauchy(n = 10, location = 0, scale = 1)
 
 (tib <- tibble(x, y, z))
 #+END_SRC
